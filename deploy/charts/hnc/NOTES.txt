1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "hnc.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "hnc.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "hnc.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "hnc.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the application status:
   kubectl get pods -l "app.kubernetes.io/name={{ include "hnc.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -n {{ .Release.Namespace }}

3. View application logs:
   kubectl logs -f deployment/{{ include "hnc.fullname" . }} -n {{ .Release.Namespace }}

4. Feature flags are configured as:
   - FEATURE_GIT: {{ .Values.env.FEATURE_GIT }}
   - FEATURE_K8S: {{ .Values.env.FEATURE_K8S }}
   - FEATURE_HHFAB: {{ .Values.env.FEATURE_HHFAB }}
   - FEATURE_GH_PR: {{ .Values.env.FEATURE_GH_PR }}

5. To update feature flags:
   helm upgrade {{ .Release.Name }} {{ .Chart.Name }} --set env.FEATURE_GIT=false -n {{ .Release.Namespace }}

6. For troubleshooting:
   - Check ConfigMap: kubectl get configmap {{ include "hnc.fullname" . }}-env -n {{ .Release.Namespace }} -o yaml
   - Describe pod: kubectl describe pod -l app.kubernetes.io/name={{ include "hnc.name" . }} -n {{ .Release.Namespace }}
   - Check events: kubectl get events -n {{ .Release.Namespace }} --sort-by=.metadata.creationTimestamp
